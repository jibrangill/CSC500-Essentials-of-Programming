#This is a simple program to perform addition and subtraction on two numbers entered by the user

#Function-1 to perform get input from user and perform the computation i.e. addition and subtraction
FUNCTION addSubtract():
    OUTPUT This is a Program to add and subtract two numbers enterd by user
    INPUT first_number
    I NPUT second_number
    FLOAT first_number
    FLOAT second_number
    SET SUM = first_number + second_number
    SET SUBTRACTION = first_number + second_number
    RETURN SUM, SUBTRACTION

#Calling the fucntion addsubtract()
result_sum, result_subtraction = CALL addSubtract()

#Print the results resturned by Function
OUTPUT result_sum
OUTPUT result_subtraction
